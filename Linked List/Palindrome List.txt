class Solution{
  public:
    //Function to check whether the list is palindrome.
    bool palindromeUtil(Node **left, Node *right){
        
        if(!right) return true;
        
        bool isP = palindromeUtil(left, right -> next);
        if(!isP) return false;
        
        isP = ((*left) -> data == right -> data);
        
        *left = (*left) -> next;
        
        return isP;
    }
    
    bool isPalindrome(Node *head)
    {
        //Your code here
        return palindromeUtil(&head, head);
    }
};

class Solution
{
    //Function to check whether the list is palindrome.
    Node reverseList(Node head){
        Node prev = head, next = null, curr = head.next;
        while(curr != null){
            next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
    
    boolean isPalindrome(Node head) 
    {
        Node slow = head, fast = head;
        while(fast.next != null && fast.next.next != null){
            slow = slow.next;
            fast = fast.next.next;
        }
        
        fast = reverseList(slow);
        
        Node p = head;
        while(fast != slow){
            if(p.data != fast.data) return false;
            p = p.next;
            fast = fast.next;
        }
        
        return true;
    }    
}
