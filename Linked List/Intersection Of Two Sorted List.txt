Node* findIntersection(Node* head1, Node* head2)
{
    // Your Code Here
    Node *intersectionHead = new Node(-1);
    Node *intersectionList = intersectionHead;
    Node *fp = head1, *sp = head2;
    while(fp && sp){
        if(fp -> data == sp -> data){
            intersectionList -> next = new Node(fp -> data);
            intersectionList = intersectionList -> next;
            fp = fp -> next;
            sp = sp -> next;
        }
        else if(fp -> data < sp -> data){
            fp = fp -> next;
        }
        else{
            sp = sp -> next;
        }
    }
    
    return intersectionHead -> next;
}

class Sol
{
   public static Node findIntersection(Node head1, Node head2)
    {
        // code here.
        Node intersectionHead = new Node(-1);
        Node intersectionList = intersectionHead;
        Node fp = head1, sp = head2;
        while(fp != null && sp != null){
            if(fp.data == sp.data){
                intersectionList.next = new Node(fp.data);
                intersectionList = intersectionList.next;
                fp = fp.next;
                sp = sp.next;
            }
            else if(fp.data < sp.data){
                fp = fp.next;
            }
            else{
                sp = sp.next;
            }
        }
        
        return intersectionHead.next;
    }
